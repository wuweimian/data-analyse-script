# -*- coding: utf-8 -*-
import pandas as pd
import report
from sklearn.preprocessing import MinMaxScaler
import genvs as GENV
G_ENV = GENV.GENVS()


"""
最大最小值调用函数
"""
def max_min_main(x_col,y_col):
    data = G_ENV.get_features_data(x_col)
    columns_col = y_col.split("|")
    scaler = MinMaxScaler()
    data_scale = data[columns_col]
    np_scaler = scaler.fit_transform(data_scale)
    pd_scaler = pd.DataFrame(np_scaler,columns=columns_col)
    # 标准化后更新原有数据
    data[columns_col] = pd_scaler
    # 保存前五行数据预览
    report.savedataframe(G_ENV.get_result_dir(), "最大最小值转换结果", data.head())
    # 保存数据
    G_ENV.update_features(data)
    # 生成报告
    report.createhtml("最大最小值转换结果", G_ENV.get_result_dir())
