# -*- coding: utf-8 -*-
import pandas as pd
from scipy.interpolate import lagrange
import report
import genvs as GENV
G_ENV = GENV.GENVS()


"""
拉格朗日插值
"""
def ployinterp_column(s, n, k=5):
    y = s[list(range(n-k, n)) + list(range(n+1, n+1+k))]
    y = y[y.notnull()]
    return lagrange(y.index, list(y))(n)

"""
拉格朗日插值调用函数
"""
def lagrange_insert_main(col,flag):
    # 获取数据
    data = G_ENV.get_features_data(col)
    #插入拉格朗日值
    if flag == "是":
        for i in data.columns:
            for j in range(len(data)):
                if(data[i].isnull())[j]:
                    data[i][j] = ployinterp_column(data[i],j)
    else:
        for i in data.columns:
            for j in range(len(data)):
                if(data[i].isnull())[j]:
                    new_col = i + "_拉格朗日插值"
                    data[new_col] = data[i]
                    data[new_col][j] = ployinterp_column(data[new_col],j)
    # 保存前五行数据预览
    report.savedataframe(G_ENV.get_result_dir(), "拉格朗日插值结果", data.head())
    # 保存数据
    G_ENV.update_features(data)
    # 生成报告
    report.createhtml("拉格朗日插值结果", G_ENV.get_result_dir())
