# -*- coding: utf-8 -*-
import pandas as pd
import report
import numpy as np
import genvs as GENV

G_ENV = GENV.GENVS()

def ManhatDistance(x,y):
    return np.sum(np.abs(x - y))

# 曼哈顿距离调用
def ManhatDistance_main(x_col):
    data = G_ENV.get_features_data(x_col)
    data = data.dropna()
    result_str="特征列|"
    result_str += x_col + "\n"
    for i in data.columns:
        result_str += i + "|"
        for j in data.columns:
            distince = ManhatDistance(data[i],data[j])
            result_str += str(distince)+"|"
        result_str = result_str[:-1]
        result_str +=  "\n"

    # 保存结果
    report.savestrtb(G_ENV.get_result_dir(), "曼哈顿距离分析", result_str)
    # 生成报告
    report.createhtml("曼哈顿距离分析", G_ENV.get_result_dir())
