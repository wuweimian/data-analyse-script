import pandas as pd
import report
from sklearn.feature_selection import SelectKBest
from sklearn.feature_selection import chi2
import genvs as GENV
G_ENV = GENV.GENVS()

"""
卡方检验调用函数
x
"""
def kafang_check_main(x,y,k=2):
    model1 = SelectKBest(chi2, k=k)
    data_np = model1.fit_transform(x, y)
    data_pd = pd.DataFrame(data_np)
    # 保存前五行数据预览
    report.savedataframe(G_ENV.get_result_dir(), "卡方检验结果", data_pd.head())
    # 生成报告
    report.createhtml("卡方检验结果", G_ENV.get_result_dir())
