# -*- coding: utf-8 -*-
import pandas as pd
import numpy as np
import report
from scipy import stats
import genvs as GENV
G_ENV = GENV.GENVS()


"""
单样本t检验调用函数
"""
def stats_main(x_col,b):
    a = G_ENV.get_features_data(x_col)
    a_mean = a.mean()

    t,p = stats.ttest_1samp(a_mean, b)
    result_str="t统计值|p统计值\n"
    result_str+="{}|{}\n".format(t,p)
    # 保存前五行数据预览
    report.savestrtb(G_ENV.get_result_dir(), "单样本t检验结果", result_str)
    # 生成报告
    report.createhtml("单样本t检验结果", G_ENV.get_result_dir())
