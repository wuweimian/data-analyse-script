# -*- coding: utf-8 -*-
import pandas as pd
import report
import genvs as GENV
G_ENV = GENV.GENVS()


"""
pearson相关性调用函数
"""
def pearson_main(x_col):
    data = G_ENV.get_features_data(x_col)
    data_corr = data.corr()
    # 保存前五行数据预览
    report.savedataframe(G_ENV.get_result_dir(), "pearson相关性结果", data_corr)
    # 生成报告
    report.createhtml("pearson相关性结果", G_ENV.get_result_dir())
