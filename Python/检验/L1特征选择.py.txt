# -*- coding: utf-8 -*-
import pandas as pd
import numpy as np
import report
from sklearn.svm import LinearSVC
from sklearn.feature_selection import SelectFromModel
from sklearn.linear_model import Lasso, Ridge
import genvs as GENV
G_ENV = GENV.GENVS()

def pretty_print_linear(coefs, names = None, sort = False):

    lst = zip(names,coefs )
    if sort:
        lst = sorted(lst,  key = lambda x:-np.abs(x[1]))
    return "\n".join("%s|%s" % (name,round(coef, 3))
                                   for name, coef in lst)

"""
L1特征选择调用函数
"""
def l1_feature(x_col,y_col):
    col = x_col + "|" + y_col
    df = G_ENV.get_features_data(col)
    col_list = x_col.split("|")
    x = (df[col_list]).values
    y = df[y_col].values

    rlasso = Lasso()
    rlasso.fit(x, y)
    result_str=""
    result_str +="特征变量|特征重要性\n"
    result_str +="{}".format(pretty_print_linear(rlasso.coef_,col_list,True))
    # 保存前五行数据预览
    report.savestrtb(G_ENV.get_result_dir(), "L1特征选择结果", result_str)
    # 生成报告
    report.createhtml("L1特征选择结果", G_ENV.get_result_dir())
